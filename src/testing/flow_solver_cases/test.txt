set(FLOW_CASES_SOURCE
    periodic_cube_flow.cpp
    1D_burgers_rewienski_snapshot.cpp
    1d_burgers_viscous_snapshot.cpp
    flow_solver_case_base.cpp)

foreach(dim RANGE 1 3)
    # Output library
    string(CONCAT FlowSolverCaseLib FlowSolverCase_${dim}D)
    add_library(${FlowSolverCaseLib} STATIC ${FLOW_CASES_SOURCE})

    string(CONCAT TestsLib Tests_${dim}D)
    target_link_libraries(${FlowSolverCaseLib} ${TestsLib})

    target_compile_definitions(${FlowSolverCaseLib} PRIVATE PHILIP_DIM=${dim})
    unset(TestsLib)

    # Setup target with deal.II
    if(NOT DOC_ONLY)
        DEAL_II_SETUP_TARGET(${FlowSolverCaseLib})
    endif()


    unset(FlowSolverCaseLib)
endforeach()
